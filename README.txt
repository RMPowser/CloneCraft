Controls:
	look around    = MOUSE X/Y
	mine		   = MOUSE LEFT
	place block    = MOUSE RIGHT
	move forward   = W
	strafe left    = A
	move backward  = S
	strafe right   = D
	jump           = SPACE
	speed modifier = SHIFT
	
	toggle flying  = F
	
	While Flying:
		move up    = SPACE
		move down  = CTRL
